\[
\begin{aligned}
  \Border{}{a}
 &= \epsilon\\
\Border{}{y \cdot a} &= \left\{
\begin{aligned}
& \Border{}{y} \cdot a 
&& \text{if} \; \Border{}{y} \cdot a \preccurlyeq y\\
& \Border{}{\Border{}{y} \cdot a} 
&& \text{otherwise}
\end{aligned}
\right.
\end{aligned}
\]
where \(x \preccurlyeq y\) means ``\(x\) is a prefix of \(y\)'', \(a\)
  is a letter and \(y\) is a non-empty word.
